CREATE TRIGGER trg_CleanupOrphanAuthors
ON Books
AFTER DELETE
AS
BEGIN
    DELETE FROM Authors
    WHERE id IN (
        SELECT d.author_id
        FROM deleted d
        WHERE NOT EXISTS (
            SELECT 1 FROM Books b WHERE b.author_id = d.author_id
        )
    );
END;
